Installation

  • npm
    $ npm install scenejs
  • es5
    <script src="//daybrush.github.io/scenejs/release/latest/scene.min.js"></script>

Make scene

import Scene from "scenejs";

const scene = new Scene({
  ".class": {
    0: "left: 0px; top: 0px"
    1: {
      "left": "100px",
      "top": "0px"
    },
    2: {
      "left": "200px",
      "top": "100px"
    }
  }
}, {
  selector: true
}).play();
